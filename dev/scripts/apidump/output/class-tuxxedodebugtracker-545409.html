Debug tracker class
This class can be used to log variables data for debugging purposes, such as iterations which can be useful for writing generators and other types of parsers to confirm states, but it is not limited to such.

Examples

Example #1
<?php

use Tuxxedo\Debug;

class Test
{
        protected $iterations     = 0;

        protected $tracker;


        public function __construct()
        {
            $this->tracker = new Debug\Tracker(function($log_data)
            {
                printf('The log was called %d time(s)%s%s', sizeof($log_data), PHP_EOL, PHP_EOL);

                if($log_data)
                {
                    foreach($log_data)
                    {
                        var_dump($log_data);
                    }
                }
            });
        }

        public function __destruct()
        {
            $this->tracker->end();
        }

        public function iterator()
        {
            $this->tracker->log(['iterations' => ++$this->iterations]);
        }
}

$test = new Test;

$test->iterator(); // Test::$iterations=1
$test->iterator(); // Test::$iterations=2

for($x = 0; $x < 40; ++$x)
{
        $test->iterator(); // Test::$iterations=N
}

unset($test);

//The log was called 42 time(s)
//
//...

?>
